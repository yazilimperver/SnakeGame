109 bool ConsoleSnakeGame::switchToGameScreen(SnakeGameScreen newScreen)
110 {
111  bool isSwitchSuccess = false;
112 
113  switch (newScreen)
114  {
115  case SnakeGameScreen::eGAME_SCREEN_LOGO:
116  if (SnakeGameScreen::eGAME_SCREEN_INITIAL == mCurrentGameScreen)
117  {
118  mCurrentGameScreen = SnakeGameScreen::eGAME_SCREEN_LOGO;
119  this->displayLogo();
120  isSwitchSuccess = true;
121  }
122  break;
123  case SnakeGameScreen::eGAME_SCREEN_SPLASH:
124  if (SnakeGameScreen::eGAME_SCREEN_LOGO == mCurrentGameScreen)
125  {
126  mCurrentGameScreen = SnakeGameScreen::eGAME_SCREEN_SPLASH;
127  this->displayGameLogo();
128  isSwitchSuccess = true;
129  }
130  break;
131  case SnakeGameScreen::eGAME_SCREEN_MENU:
132  if (SnakeGameScreen::eGAME_SCREEN_SPLASH == mCurrentGameScreen
133  ||
134  SnakeGameScreen::eGAME_SCREEN_GAMEOVER == mCurrentGameScreen)
135  {
136  clearConsole();
137 
139  setColor(Color::eColor_black, Color::eColor_green);
140  LogoDisplayer::DisplaySnakeLogo();
141 
142  mMainMenu.reset();
143 
144  mCurrentGameScreen = SnakeGameScreen::eGAME_SCREEN_MENU;
145  isSwitchSuccess = true;
146  }
147  break;
148  case SnakeGameScreen::eGAME_SCREEN_GAME:
149  if (SnakeGameScreen::eGAME_SCREEN_MENU == mCurrentGameScreen
150  ||
151  SnakeGameScreen::eGAME_SCREEN_SPLASH == mCurrentGameScreen)
152  {
153  mFruitGenerator.reset();
154 
155  mCurrentGameScreen = SnakeGameScreen::eGAME_SCREEN_GAME;
156  this->prepareGameScreen();
157  mGameStartClock = chrono::high_resolution_clock::now();
158  isSwitchSuccess = true;
159  mMidiPlayer.playMusic(".\\Data\\music.mid");
160  }
161  else if (SnakeGameScreen::eGAME_SCREEN_GAMEOVER == mCurrentGameScreen)
162  {
163  mFruitGenerator.reset();
164 
165  mCurrentGameScreen = SnakeGameScreen::eGAME_SCREEN_GAME;
166  clearConsole();
167  this->prepareGameScreen();
168  mGameStartClock = chrono::high_resolution_clock::now();
169  isSwitchSuccess = true;
170  mMidiPlayer.playMusic(".\\Data\\music.mid");
171  }
172  break;
173  case SnakeGameScreen::eGAME_SCREEN_GAMEOVER:
174  if (SnakeGameScreen::eGAME_SCREEN_GAME == mCurrentGameScreen)
175  {
176  mCurrentGameScreen = SnakeGameScreen::eGAME_SCREEN_GAMEOVER;
177  isSwitchSuccess = true;
178  mMidiPlayer.stopMusic();
179  }
180  break;
181  default:
182  break;
183  }
184 
185  if (false == isSwitchSuccess)
186  {
187  setColor(Color::eColor_red, Color::eColor_yellow);
188  displayCenteredText("Unexpected game screen change request!", 126, 38);
189  }
190 
191  return isSwitchSuccess;
192 }

194 void ConsoleSnakeGame::displayGameOver(float tickTime)
195 {

200

206  elapsed = 0;

208 
209  if (true == displayText)
210  {
211  displayCenteredText("GAME OVER!", 126, 15);
212  displayCenteredText("Please Press Space Bar to return menu!", 126, 36);
213 
214  // TODO: Also there are alive players!

216  {
217  displayCenteredText("Please Press Enter to continue watch game!", 126, 37);
218  }
219  }
220  else
221  {
222  displayCenteredText(" ", 126, 15);
223  }
224 
225  if (getNonBlockingChar() == ' ')
226  {
227  elapsed = 0.0F;
228  displayText = true;
229 
230  // restart game
231  this->switchToGameScreen(SnakeGameScreen::eGAME_SCREEN_MENU);
232  }
